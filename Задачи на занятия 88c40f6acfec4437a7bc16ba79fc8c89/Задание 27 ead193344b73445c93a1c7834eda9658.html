file = open('Files/27/3/27_B_3.txt')

k = int(file.readline())
n = int(file.readline())

nums = [int(i) for i in file.readlines()]
mins = [nums[i] for i in range(k)]
maxs = [nums[i] for i in range(k)]
res = []

for i in range(k, n):
	res.append(nums[i] * mins[i % k])
	res.append(nums[i] * maxs[i % k])
	mins[i % k] = min(mins[i % k], nums[i])
	maxs[i % k] = max(maxs[i % k], nums[i])

print(max(res))

file = open('Files/27/6/27_6B.txt')
k = int(file.readline())
n = int(file.readline())
nums = [int(i) for i in file.readlines()]

max_odd = -1
max_even = -1
res = []

for i in range(n - k - 1, 0, -1):
	if nums[i + k] % 2 == 0:
		max_even = max(max_even, nums[i + k])
	else:
		max_odd = max(max_odd, nums[i + k])
	if nums[i] % 2 == 0:
		res.append(nums[i] + max_odd)
	else:
		res.append(nums[i] + max_even)

print(max(res))
