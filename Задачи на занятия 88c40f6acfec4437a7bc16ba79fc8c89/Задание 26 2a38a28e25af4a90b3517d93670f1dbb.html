import math

file = open('Files/26/26_1.txt', 'r')
n = int(file.readline())

with_sale = []
no_sale = []

for i in file.readlines():
	if int(i) > 350:
		with_sale.append(int(i))
	else:
		no_sale.append(int(i))

with_sale.sort()
one_check = sum(no_sale) + math.ceil(sum(i * 0.25 for i in with_sale[:len(with_sale) // 3]) + sum(with_sale[len(with_sale) // 3:]))
with_sale.sort(reverse=True)
many_bills = 0

for i in range(len(with_sale)):
	if i % 3 == 2:
		many_bills += math.ceil(with_sale[i] * 0.25) + with_sale[i - 1] + with_sale[i-2]

print(many_bills + sum(no_sale), one_check)

file = open('Files/26/26_2.txt', 'r')
n = int(file.readline())

times = [[int(i) for i in j.split()] for j in file.readlines()]
times.sort(key=lambda x: x[0])

min_start = times[0][0]
max_end = times[0][1]

times.append([100000, 100000])
segments = []

for start, end in times:
	if max_end < start:
		segments.append(max_end - min_start)
		min_start = start
	max_end = max(end, max_end)

print(times[-10:])
print(sum(segments), max(segments))

for time in times:
	if chosen[-1][1] + 20 <= time[0]:
		chosen.append(time)

last_start = max([s for s, e in times if s >= last_end + 20])

file = open('Files/26/26_4.txt', 'r')
k, m, n = [int(i) for i in file.readline().split()]

orders = [int(i) for i in file.readlines()]
free_seats = [m] * k
orders.sort(reverse=True)

count = 0
for order in orders:
	for i in range(k):
		if free_seats[i] >= order:
			free_seats[i] -= order
			count += 1
			break

print(count, sum(free_seats))

file = open('Files/26/26_5.txt')
n = int(file.readline())

orders = [[int(j) for j in i.split()] for i in file.readlines()]
orders.sort()
max_row = -1

count = 0

for i in range(1, n - 1):
	if orders[i - 1][0] == orders[i][0] == orders[i + 1][0] *and* orders[i][1] - orders[i - 1][1] == 6 == orders[i + 1][1] - orders[i][1]:
		max_row = max(max_row, orders[i][0])
		count += 1

print(max_row, count)
